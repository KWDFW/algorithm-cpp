[707设计链表](https://leetcode.cn/problems/design-linked-list/)

## 思路
1、通过设置虚拟头节点来避免需要特殊处理头节点

2、需要初始化链表，设置链表的长度和虚拟头节点

3、不要让方法中可能出现null->next，可以某个节点的next指向null

4、注意cur的初始位置，需要对头结点操作时，应该指向虚拟头节点;不需要对头结点操作时，应指向头结点

## 代码
```cpp
class MyLinkedList {
public:
    //定义链表节点  
    struct LinkedNode {
        int val;
        LinkedNode* next;
        LinkedNode(int val):val(val), next(nullptr){}
    };

    //初始化链表
    MyLinkedList() {   
        _dummyhead = new LinkedNode(0);
        _size = 0;
    }
  
    int get(int index) {
        if (index > (_size - 1) || index < 0) {
            return -1;
        }
        LinkedNode* cur = _dummyhead->next;
        while (index--) {
            cur = cur->next;
        }
        // printLinkedlist("get");
        return cur->val;
    }
    
    void addAtHead(int val) {
        LinkedNode* newNode = new LinkedNode(val);
        newNode->next = _dummyhead->next;
        _dummyhead->next = newNode;
        _size++;
        // printLinkedlist("head");
    }
    
    void addAtTail(int val) {
        LinkedNode* newNode = new LinkedNode(val);
        LinkedNode* cur = _dummyhead;
        while (cur->next != nullptr) {
            cur = cur->next;
        }
        cur->next = newNode;
        _size++;
        // printLinkedlist("tail");
    }
    
    void addAtIndex(int index, int val) {
        if (index > _size ) {
            return;
        }
        if (index <= 0) {
            addAtHead(val);
            return;
        }
        if (index == _size) {
            addAtTail(val);
            return;
        }
        LinkedNode* cur = _dummyhead;
        while(index) {
            cur = cur->next;
            index--;
        }   
        LinkedNode* newNode = new LinkedNode(val);
        newNode->next = cur->next;
        cur->next = newNode;
        _size++;
        // printLinkedlist("add");
    }
    
    void deleteAtIndex(int index) {
        if (index < 0 || index > (_size - 1)) {
            return;
        }
        LinkedNode* cur = _dummyhead;
        while (index--) {
            cur = cur->next;
        }
        LinkedNode* temp = cur->next;
        cur->next = cur->next->next;
        delete temp;
        _size--;
        // printLinkedlist("del");
    }

    void printLinkedlist(string a) {
        LinkedNode* cur = _dummyhead;
        while (cur->next != nullptr) {
            cout << cur->next->val << " ";
            cur = cur->next;
        } 
        cout << a << " ";
        cout << "size:" << _size;
        cout << endl;
    }
private:
    int _size;
    LinkedNode* _dummyhead;
};

/**
 * Your MyLinkedList object will be instantiated and called as such:
 * MyLinkedList* obj = new MyLinkedList();
 * int param_1 = obj->get(index);
 * obj->addAtHead(val);
 * obj->addAtTail(val);
 * obj->addAtIndex(index,val);
 * obj->deleteAtIndex(index);
 */
```
## 复杂度分析
时间复杂度：get():O(index), head():O(1), tail():O(n), add:O(index), delete:O(index)

空间复杂度：单个函数调用：O(1)。整体函数调用：O(n)，n为调用次数之和
